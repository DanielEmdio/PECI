CREATE TABLE Users(
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
    user TEXT,
    password TEXT,
    PTid INTEGER,
    FOREIGN KEY (PTid) REFERENCES PTs(Id)
);


CREATE TABLE PTs(
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
    user TEXT,
    password TEXT,
    video TEXT,
    restrictedVideo INTEGER
);
